## Examples

### Basic Usage

```javascript
import React from 'react';
import { GAIKProvider, useInsights } from 'gaik-toolkit';

function App() {
  return (
    <GAIKProvider apiKey="your-key">
      <BusinessDashboard />
    </GAIKProvider>
  );
}

function BusinessDashboard() {
  const { insights, loading } = useInsights(businessData);
  
  if (loading) return <div>Generating insights...</div>;
  
  return (
    <div>
      {insights.map(insight => (
        <AIInsightCard key={insight.id} insight={insight} />
      ))}
    </div>
  );
}
```

### Advanced Integration

```javascript
// Custom hook for GAIK integration
import { useGAIK } from 'gaik-toolkit';

function useBusinessInsights(data) {
  const { client } = useGAIK();
  
  return useQuery(['insights', data], () => 
    client.generateInsights(data)
  );
}
```
